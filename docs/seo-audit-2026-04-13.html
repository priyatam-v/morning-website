Schema Implementation (Copy-Ready)

Add this single <script> block to app/layout.tsx. It declares WebSite + Organization + SoftwareApplication in one @graph.

<script type="application/ld+json">
{
  "@context": "https://schema.org",
  "@graph": [
    {
      "@type": "WebSite",
      "@id": "https://www.getmorning.co/#website",
      "name": "Morning",
      "url": "https://www.getmorning.co",
      "description": "A daily knowledge brief delivering ten curated knowledge cards every day.",
      "inLanguage": "en-US",
      "publisher": { "@id": "https://www.getmorning.co/#organization" }
    },
    {
      "@type": "Organization",
      "@id": "https://www.getmorning.co/#organization",
      "name": "Morning",
      "url": "https://www.getmorning.co",
      "logo": {
        "@type": "ImageObject",
        "url": "https://www.getmorning.co/icon.png",
        "width": 442,
        "height": 493
      },
      "description": "Morning delivers ten high-quality knowledge cards every day — the antidote to doomscrolling.",
      "foundingDate": "2026",
      "contactPoint": { "@type": "ContactPoint", "email": "hello@getmorning.co" },
      "sameAs": ["https://www.instagram.com/REPLACE_WITH_HANDLE"]
